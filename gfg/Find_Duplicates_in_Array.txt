class Solution {
    public static ArrayList<Integer> duplicates(int arr[], int n) {
        Arrays.sort(arr);
        int count = 0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]== arr[i+1])
            {
                count+=1;
            }
            else
            {
                break;
            }
        if(count>1){
            return arr[i];
        }
        }
        return -1;
    }
}
